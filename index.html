import express from "express";
import cors from "cors";
import helmet from "helmet";
import jwt from "jsonwebtoken";

const app = express();
app.use(express.json());
app.use(cors({ origin: "*" }));
app.use(helmet());

const JWT_SECRET = process.env.JWT_SECRET || "CHANGE_ME_NOW";

function auth(req,res,next){
  const h = req.headers.authorization || "";
  const token = h.startsWith("Bearer ") ? h.slice(7) : "";
  try{
    req.user = jwt.verify(token, JWT_SECRET);
    next();
  }catch(e){
    return res.status(401).json({ error: "Unauthorized" });
  }
}

// ---- AUTH
app.post("/auth/login", (req,res) => {
  const { email } = req.body || {};
  // Demo login: replace with real user store
  const token = jwt.sign({ email, role:"admin" }, JWT_SECRET, { expiresIn: "8h" });
  res.json({ token });
});

// ---- PROJECTS (demo in-memory)
const db = { projects: [], milestones: [], audit: [] };
const log = (type, data) => db.audit.push({ ts: new Date().toISOString(), type, data });

app.post("/projects", auth, (req,res) => {
  const p = { id: "P"+(db.projects.length+1), ...req.body };
  db.projects.push(p); log("PROJECT_CREATE", p);
  res.json(p);
});

app.post("/milestones", auth, (req,res) => {
  const m = { id: "M"+(db.milestones.length+1), ...req.body };
  db.milestones.push(m); log("MILESTONE_CREATE", m);
  res.json(m);
});

// ---- ESCROW (state machine demo)
app.post("/escrow/deposit", auth, (req,res) => {
  const { projectId, amount, currency } = req.body || {};
  // Here you would call licensed partner API (bank/EMI/PSP) and store reference
  const dep = { projectId, amount, currency, status:"LOCKED", ref:"DEP-"+Date.now() };
  log("ESCROW_DEPOSIT", dep);
  res.json(dep);
});

  const { projectId, milestoneId, amount, currency } = req.body || {};
  // Here you would verify milestone (docs + expert + AI) then call partner API release
